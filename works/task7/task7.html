JavaScript Assignment and Arithmetic Operations

Open your browser's developer console to see the output.


// Assignment operations
let a = 10;   // Simple assignment
console.log(a);
a += 5;  // Addition assignment
console.log(a);
a -= 3;  // Subtraction assignment
console.log(a);
a *= 2;  // Multiplication assignment
console.log(a);
a /= 4;  // Division assignment
console.log(a);

// Arithmetic operations
let x = 15;
let sum = x + 5;   // Addition
console.log(sum);

let difference = x - 3;  // Subtraction
console.log(difference);

let product = x * 2;  // Multiplication
console.log(product);
